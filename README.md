# clipSphere

## 安装和运行
- npm install
- npm run dev

## 目录结构
├── src
│   ├── api                    #接口
│   ├── assets                 # 图片、icon
│   ├── components             # 全局公共组件
│   ├── constants              # 全局公共常量
│   └── hooks
│       ├── useBasicLayout     # 判断现在屏幕尺寸
│       ├── useIconRender      # 页面用icon
│       ├── useLanguage        # 当前国际化
│       └── useTheme           # 主题
│   ├── layout                 # 公共布局
│   ├── loading                # 页面加载
│   ├── locales                # 国际化
│   ├── plugins                # 插件公共样式
│   ├── router                 # 路由
│   ├── store                  # 跨组件/页面共享状态存储库
│   ├── styles                 # 插件公共样式
│   ├── typings                # ts公共interface
│   ├── utils                  # 公共方法
│   └── views                  # 页面
├── package.json               # npm配置文件
└── README.md                  # 项目的说明文档
